glob { 
  unused
} 

proc { 
} 

func { 
  absolute (n) {
    local { output }
    if ( n > 0 ) {
      output = n
    } else {
      output = ( neg n )
    } ;
    return output
  }
  
  maximum (a b) {
    local { output }
    if ( a > b ) {
      output = a
    } else {
      output = b
    } ;
    return output
  }
  
  sign (n) {
    local { output }
    if ( n > 0 ) {
      output = 1
    } else {
      if ( n eq 0 ) {
        output = 0
      } else {
        output = ( neg 1 )
      }
    } ;
    return output
  }
} 

main { 
  var { result temp }
  
  print "FUNC COND" ;
  
  print "Absolute pos" ;
  result = absolute ( 15 ) ;
  print result ;
  
  print "Absolute neg" ;
  temp = ( neg 15 );
  result = absolute ( temp ) ;
  print result ;

  print "Maximum" ;
  result = maximum ( 10 25 ) ;
  print result ;
  
  print "Sign pos" ;
  result = sign ( 10 ) ;
  print result ;

  print "Sign zero" ;
  result = sign ( 0 ) ;
  print result ;

  print "Sign neg" ;
  temp = ( neg 10 ) ;
  result = sign ( temp ) ;
  print result ;
  
  print "DONE" ;
  halt
}
