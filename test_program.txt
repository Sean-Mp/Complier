// Sample SPL Program
glob { x y z }
proc {
  printnum(n) {
    local { }
    print n
  }
}
func {
  add(a b) {
    local { result }
    result = (a plus b);
    return result
  }
}
main {
  var { num1 num2 sum }
  num1 = 5;
  num2 = 10;
  sum = add(num1 num2);
  print "Result";
  printnum(sum);
  halt
}
