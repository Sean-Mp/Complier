glob { 
  globalvar
} 

proc { 
  noparams () {
    local {}
    print "No params"
  }
  
  oneparam (x) {
    local {}
    print "One param" ;
    print x
  }
  
  twoparams (a b) {
    local {}
    print "Two params" ;
    print a ;
    print b
  }
  
  threeparams (x y z) {
    local { sumval }
    sumval = ( ( x plus y ) plus z ) ;
    print "Three" ;
    print sumval
  }
  
  modifyglobal () {
    local {}
    print "Before mod:" ;
    print globalvar ;
    globalvar = 999 ;
    print "After mod:" ;
    print globalvar
  }
} 

func { 
  addtwo (a b) {
    local { result }
    result = ( a plus b ) ;
    print "Adding numbers:" ;
    print result ;
    return result
  }
} 

main { 
  var { a b temp1 temp2 }

  print "PROCEDURES" ;

  print "No params" ;
  noparams ( ) ;

  print "One param" ;
  oneparam ( 42 ) ;

  print "Two params" ;
  twoparams ( 10 20 ) ;

  print "Three params" ;
  threeparams ( 1 2 3 ) ;

  print "Expr args" ;
  a = 5 ;
  b = 3 ;
  temp1 = ( a plus b ) ;
  temp2 = ( a mult b ) ;
  twoparams ( temp1 temp2 ) ;

  print "Global mod" ;
  globalvar = 100 ;
  print globalvar ;
  modifyglobal ( ) ;
  print globalvar ;

  print "Function test" ;
  a = addtwo ( 4 6 ) ;
  print "Sum result:" ;
  print a ;

  print "DONE" ;
  halt
}
